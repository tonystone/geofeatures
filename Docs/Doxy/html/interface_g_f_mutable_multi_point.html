Parameters

Example:

{
GFMutableMultiPoint * multiPoint = [[GFMutableMultiPoint alloc] init];
multiPoint[0] = [[GFPoint alloc] initWithWKT: @"POINT(1 1)"];
multiPoint[1] = [[GFPoint alloc] initWithWKT: @"POINT(2 2)"];
}
